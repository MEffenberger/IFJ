161void symtable_insert(AVL_tree **tree, char *key, sym_data *data) {
162 if (*tree == NULL) { // insert first to an empty tree
163 *tree = (AVL_tree *)allocate_memory(sizeof(AVL_tree));
164 (*tree)->key = key;
165 (*tree)->data = data;
166 (*tree)->left = NULL;
167 (*tree)->right = NULL;
168 (*tree)->height = 0;
169 (*tree)->nickname = 0;
170 }
171 else if (strcmp((*tree)->key, key) > 0) {
172 symtable_insert(&((*tree)->left), key, data);
173 }
174 else if (strcmp((*tree)->key, key) < 0) {
175 symtable_insert(&((*tree)->right), key, data);
176 }
177 else { // strcmp((*tree)->key, key) == 0
178 error_exit(ERROR_SEM_UNDEF_FUN, "SYMTABLE", "Key already exists in the symbol table.");
179 }
180
181 // correct the height and continue with balancing
182 (*tree)->height = 1 + MAX(height((*tree)->left), height((*tree)->right));
183
184 // check if the tree is balanced: 0 balanced, -1 left higher, 1 right higher
185 int balance = get_balance(*tree);
186 if (balance != 0) { // if the tree is not balanced
187
188 // LL case
189 if (balance > 1 && strcmp((*tree)->key, key) > 0) {
190 right_rotate(tree);
191 }
192 // RR case
193 else if (balance < -1 && strcmp((*tree)->key, key) < 0) {
194 left_rotate(tree);
195 }
196 // LR case
197 else if (balance > 1 && strcmp((*tree)->key, key) < 0) {
198 left_rotate(&((*tree)->left));
199 right_rotate(tree);
200 }
201 // RL case
202 else if (balance < -1 && strcmp((*tree)->key, key) > 0) {
203 right_rotate(&((*tree)->right));
204 left_rotate(tree);
205 }
206 }

210void symtable_delete(AVL_tree **tree, char *key) {
211 if ((*tree) != NULL) {
212 if (strcmp((*tree)->key, key) > 0) {
213 symtable_delete(&((*tree)->left), key);
214 }
215 else if (strcmp((*tree)->key, key) < 0) {
216 symtable_delete(&((*tree)->right), key);
217 }
218 else { // strcmp((*tree)->key, key) == 0 -> found the node to delete
219 if ((*tree)->left == NULL && (*tree)->right == NULL) { // no children
220 free(*tree);
221 *tree = NULL;
222 }
223 else if ((*tree)->left != NULL && (*tree)->right != NULL) { // both children
224 AVL_tree *tmp = (*tree)->right;
225 while (tmp->left != NULL) {
226 tmp = tmp->left;
227 }
228 (*tree)->key = tmp->key;
229 (*tree)->data = tmp->data;
230 symtable_delete(&((*tree)->right), tmp->key);
231 }
232 else { // only one child
233 AVL_tree *tmp = *tree;
234 if ((*tree)->left == NULL) { // only right child
235 *tree = (*tree)->right;
236 }
237 else if ((*tree)->right == NULL) { // only left child
238 *tree = (*tree)->left;
239 }
240 free(tmp);
241 }
242 }
243
244 // if the tree is not empty, correct the height and continue with balancing
245 if (*tree != NULL) {
246 (*tree)->height = 1 + MAX(height((*tree)->left), height((*tree)->right));
247
248 // check if the tree is balanced: 0 balanced, -1 left higher, 1 right higher
249 int balance = get_balance(*tree);
250 if (balance != 0) { // if the tree is not balanced
251
252 // LL case
253 if (balance > 1 && get_balance((*tree)->left) >= 0) {
254 right_rotate(tree);
255 }
256 // RR case
257 else if (balance < -1 && get_balance((*tree)->right) <= 0) {
258 left_rotate(tree);
259 }
260 // LR case
261 else if (balance > 1 && get_balance((*tree)->left) < 0) {
262 left_rotate(&((*tree)->left));
263 right_rotate(tree);
264 }
265 // RL case
266 else if (balance < -1 && get_balance((*tree)->right) > 0) {
