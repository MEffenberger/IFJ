89void inst_list_delete_after(instruction_list *list) {
90 instruction *inst = list->active->next;
91 // re-pointing the pointers across the deleted node
92 if (inst == list->last) {
93 list->last = list->active;
94 }
95 else {
96 inst->next->prev = list->active;
97 }
98 list->active->next = inst->next;
99 free(inst);
100}

102void inst_list_dispose(instruction_list *list) {
103 instruction* inst = list->first;
104 while (list->first != NULL) {
105 list->first = list->first->next;
106 free(inst);
107 inst = list->first;
108 }

110 list->active = NULL;
111 list->first = NULL;
112 list->last = NULL;

360// function definition
361void codegen_func_def(instruction *inst) {
362 printf("JUMP !!skip_%s\n", inst->name);
363 printf("LABEL %s\n", inst->name);
364 printf("PUSHFRAME\n");
365 printf("DEFVAR LF@$retval$\n");
366 for (int i = 1; i <= inst->cnt; i++) {
367 // find the parameter in the symtable based on its param_order
368 AVL_tree* param = symtable_find_param(inst->relevant_node->symtable, i);
369 printf("DEFVAR LF@%s\n", param->key);
370 printf("MOVE LF@%s LF@$%d\n", param->key, i);
371 }
372}

521void codegen_substring(instruction *inst) {
522 printf("JUMP !!skip_substring\n");
523 printf("LABEL substring\n");
524 printf("PUSHFRAME\n");
525 printf("DEFVAR LF@$retval$\n");
526 printf("MOVE LF@$retval$ string@\n");
527 printf("DEFVAR LF@$tmp1\n");
528 printf("DEFVAR LF@$check\n");
529 printf("MOVE LF@$check bool@false\n"); // retval is nil if:
530 printf("LT LF@$check LF@$2 int@0\n"); // startingAt < 0
531 printf("JUMPIFEQ !!load_nil LF@$check bool@true\n");
532 printf("LT LF@$check LF@$3 int@0\n"); // endingBefore < 0
533 printf("JUMPIFEQ !!load_nil LF@$check bool@true\n");
534 printf("GT LF@$check LF@$2 LF@$3\n"); // startingAt > endingBefore
535 printf("JUMPIFEQ !!load_nil LF@$check bool@true\n");
536 printf("EQ LF@$check LF@$2 LF@$3\n");
537 printf("JUMPIFEQ !!load_result LF@$check bool@true\n");
538 printf("DEFVAR LF@$tmp_strlen\n");
539 printf("STRLEN LF@$tmp_strlen LF@$1\n");
540 printf("GT LF@$check LF@$2 LF@$tmp_strlen\n"); // startingAt > strlen
541 printf("JUMPIFEQ !!load_nil LF@$check bool@true\n");
542 printf("EQ LF@$check LF@$2 LF@$tmp_strlen\n"); // startingAt == strlen
543 printf("JUMPIFEQ !!load_nil LF@$check bool@true\n");
544 printf("GT LF@$check LF@$3 LF@$tmp_strlen\n"); // endingBefore > strlen
545 printf("JUMPIFEQ !!load_nil LF@$check bool@true\n");
546 printf("LABEL !!substring_loop\n");
547 printf("GETCHAR LF@$tmp1 LF@$1 LF@$2\n");
548 printf("CONCAT LF@$retval$ LF@$retval$ LF@$tmp1\n");
549 printf("ADD LF@$2 LF@$2 int@1\n");
550 printf("JUMPIFNEQ !!substring_loop LF@$2 LF@$3\n");
551 printf("JUMP !!load_result\n");
552 printf("LABEL !!load_nil\n"); // load nil if any of the conditions above is true
553 printf("MOVE LF@$retval$ nil@nil\n");
554 printf("LABEL !!load_result\n"); // load result, if startingAt == endingBefore, it's empty string
555 printf("POPFRAME\n");
556 printf("RETURN\n");
