46void forest_insert(forest_node *parent, f_keyword_t keyword, char *name, forest_node **active) {
47 if (parent != NULL) {

49 child->name = name;

58
59 if (parent->children == NULL) {
60 parent->children = (forest_node**)allocate_memory((sizeof(forest_node*)));
61 } else {
62 // Resize the existing array to accommodate the new child
63 parent->children = (forest_node**)realloc(parent->children, (parent->children_count + 1) * sizeof(forest_node*));
64 if (parent->children == NULL) {
65 error_exit(ERROR_INTERNAL, "REALLOC", "Memory reallocation for forest node children failed");
66 }
67 }
68 parent->children[parent->children_count] = child;
69 parent->children_count++;
70 child->symtable = NULL;
71 *active = child;
